CREATE OR REPLACE TRIGGER "ATLAS".BEF_TECNICO_ACESSO_INDICE BEFORE
   INSERT OR
   UPDATE ON ATLAS.TECNICO_ACESSO FOR EACH ROW DECLARE SEQ NUMBER;
  BEGIN
    IF NVL(:NEW.SEQ,0) = 0 THEN
       SELECT NVL(MAX(SEQ),0)+1 INTO SEQ FROM ATLAS.TECNICO_ACESSO;
      
      :NEW.SEQ := SEQ;
    END IF;
  END;
